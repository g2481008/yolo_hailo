# ROS + 仮想環境エイリアス
alias venv_ros='\
  [ -z "$VIRTUAL_ENV" ] && source ~/hailo_examples/degirum_env/bin/activate; \
  export PYTHONPATH="/root/librealsense/build/Release:$PYTHONPATH"; \
  source /opt/ros/humble/setup.bash; \
  source ~/ros2_ws/install/setup.bash; \
  export LD_LIBRARY_PATH="/usr/local/lib${LD_LIBRARY_PATH:+:$LD_LIBRARY_PATH}" 
'

# rqt_image_view 起動
alias rqt='\
  source .bashrc && \
  source /opt/ros/humble/setup.bash && \
  source ~/ros2_ws/install/setup.bash && \
